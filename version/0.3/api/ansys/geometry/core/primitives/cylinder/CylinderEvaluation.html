CylinderEvaluation#

class CylinderEvaluation(cylinder: Cylinder, parameter: ansys.geometry.core.primitives.parameterization.ParamUV)#

Bases: ansys.geometry.core.primitives.surface_evaluation.SurfaceEvaluation
